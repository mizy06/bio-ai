最小爬取伪代码

1. 从 sitemap 或 archive 枚举 article URL。
2. 进入 article 页面后提取 title、authors、abstract、methods、ethics、timeline 等栏目。
3. 把 DOI、publication date、subject tag 和 section 结构分开保存。
4. 对开放页面直接抓取全文 HTML。
5. 如后续需要下载 PDF，再从 article 页补充文件资产层。
爬取脚本
from pathlib import Path
import sys

sys.path.append(str(Path(__file__).resolve().parents[1]))
from common import fetch_text, parse_xml_loc_tags, save_snapshot


def main() -> None:
    sitemap = fetch_text("https://www.researchprotocols.org/sitemap.xml")
    urls = parse_xml_loc_tags(sitemap.get("text", ""), 20)
    payload = {"sample_urls": urls[:10]}
    print(save_snapshot("wet/jmir_research_protocols.json", payload))


if __name__ == "__main__":
    main()
