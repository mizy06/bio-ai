class RateLimiter:
    def __init__(self, max_rps: float) -> None:
        self.interval = 0.0 if max_rps <= 0 else 1.0 / max_rps
        self.lock = threading.Lock()
        self.next_allowed_at = 0.0

    def wait(self) -> None:
        if self.interval <= 0:
            return
        while True:
            with self.lock:
                now = time.monotonic()
                if now >= self.next_allowed_at:
                    self.next_allowed_at = now + self.interval
                    return
                sleep_for = self.next_allowed_at - now
            time.sleep(sleep_for)


def stable_json_bytes(payload: object) -> bytes:
    return json.dumps(payload, ensure_ascii=False, sort_keys=True, separators=(",", ":")).encode("utf-8")


def write_blob_pair(abspath: Path, data: bytes, compressed: bytes, materialize_uncompressed: bool) -> None:
    if materialize_uncompressed:
        plain_path = Path(str(abspath)[:-3]) if str(abspath).endswith(".gz") else abspath
        plain_path.parent.mkdir(parents=True, exist_ok=True)
        if not plain_path.exists():
            tmp_plain_path = plain_path.with_name(f"{plain_path.name}.tmp.{os.getpid()}")
            tmp_plain_path.write_bytes(data)
            tmp_plain_path.replace(plain_path)
        if abspath != plain_path and abspath.exists():
            abspath.unlink()
    else:
        abspath.parent.mkdir(parents=True, exist_ok=True)
        if not abspath.exists():
            tmp_path = abspath.with_name(f"{abspath.name}.tmp.{os.getpid()}")
            tmp_path.write_bytes(compressed)
            tmp_path.replace(abspath)

def store_json_blob(
    archive_root: Path,
    blob_kind: str,
    payload: object,
    materialize_uncompressed: bool = DEFAULT_MATERIALIZE_UNCOMPRESSED,
) -> dict[str, object]:
    data = stable_json_bytes(payload)
    sha256 = hashlib.sha256(data).hexdigest()
    filename = f"{sha256}.json" if materialize_uncompressed else f"{sha256}.json.gz"
    relpath = Path("objects") / "json" / blob_kind / filename
    abspath = archive_root / relpath
    compressed = gzip.compress(data, compresslevel=6)
    write_blob_pair(abspath, data, compressed, materialize_uncompressed)
    return {
        "sha256": sha256,
        "storage_relpath": str(relpath),
        "materialized_relpath": str(relpath) if materialize_uncompressed else None,
        "blob_kind": blob_kind,
        "byte_length": len(data),
        "gzip_length": len(compressed),
    }


def store_text_blob(
    archive_root: Path,
    text: str,
    source_format: str,
    materialize_uncompressed: bool = DEFAULT_MATERIALIZE_UNCOMPRESSED,
) -> dict[str, object]:
    data = text.encode("utf-8")
    sha256 = hashlib.sha256(data).hexdigest()
    filename = f"{sha256}.txt" if materialize_uncompressed else f"{sha256}.txt.gz"
    relpath = Path("objects") / "text" / filename
    abspath = archive_root / relpath
    compressed = gzip.compress(data, compresslevel=6)
    write_blob_pair(abspath, data, compressed, materialize_uncompressed)
    return {
        "sha256": sha256,
        "storage_relpath": str(relpath),
        "materialized_relpath": str(relpath) if materialize_uncompressed else None,
        "source_format": source_format,
        "byte_length": len(data),
        "gzip_length": len(compressed),
        "char_length": len(text),
        "line_count": text.count("\n") + (1 if text else 0),
    }

def collapse_ws(text: str | None) -> str:
    if not text:
        return ""
    return SPACE_RE.sub(" ", text).strip()


def ordered_unique(items: list[str]) -> list[str]:
    seen: set[str] = set()
    result: list[str] = []
    for item in items:
        if item and item not in seen:
            seen.add(item)
            result.append(item)
    return result


def maybe_parse_json_text(value: object) -> object | None:
    if not isinstance(value, str):
        return None
    stripped = value.strip()
    if not stripped or stripped[0] not in "{[":
        return None
    try:
        return json.loads(stripped)
    except json.JSONDecodeError:
        return None


def is_draft_rich_text_doc(value: object) -> bool:

def draft_entity_summary(entity: object) -> str:
    if not isinstance(entity, dict):
        return ""
    entity_type = collapse_ws(str(entity.get("type") or "")).lower()
    data = entity.get("data")
    if not isinstance(data, dict):
        data = {}

    if entity_type in {"notes", "note"}:
        note = decode_protocols_rich_text(data)
        return f"Note: {note}" if note else ""

    if entity_type == "software":
        name = collapse_ws(str(data.get("name") or ""))
        version = collapse_ws(str(data.get("version") or ""))
        details = name
        if name and version:
            details = f"{name} {version}"
        elif version:
            details = version
        return f"Software: {details}" if details else ""

    if entity_type == "citation":
        title = collapse_ws(str(data.get("title") or ""))
        doi = collapse_ws(str(data.get("doi") or ""))
        details = title or doi
        return f"Citation: {details}" if details else ""

    if entity_type == "file":
        name = collapse_ws(str(data.get("original_name") or data.get("file_name") or ""))
        source = collapse_ws(str(data.get("source") or ""))
        details = name or source
        return f"Attachment: {details}" if details else ""

    for field in ("title", "name", "label", "value", "text", "source", "link"):
        candidate = collapse_ws(str(data.get(field) or ""))
        if candidate:
            prefix = entity_type.replace("_", " ").title() if entity_type else "Entity"
            return f"{prefix}: {candidate}"
    return ""


def render_draft_rich_text(doc: dict[str, object]) -> str:
    entity_map = doc.get("entityMap")
    if not isinstance(entity_map, dict):
        entity_map = {}

    parts: list[str] = []
    for block in doc.get("blocks") or []:
        if not isinstance(block, dict):
            continue
        text = collapse_ws(html.unescape(str(block.get("text") or "")))
        entity_ranges = block.get("entityRanges")
        entity_lines: list[str] = []
        if isinstance(entity_ranges, list):
            for entity_range in entity_ranges:
                if not isinstance(entity_range, dict):
                    continue
                entity_key = str(entity_range.get("key"))
                summary = draft_entity_summary(entity_map.get(entity_key))
                if summary and summary not in entity_lines and summary not in text:
                    entity_lines.append(summary)

        block_lines: list[str] = []
        if text:
            block_lines.append(text)
        if entity_lines and (not text or collapse_ws(str(block.get("type") or "")).lower() == "atomic"):
            block_lines.extend(entity_lines)
        if block_lines:
            parts.append("\n".join(block_lines))

    return BLANKS_RE.sub("\n\n", "\n\n".join(parts).strip())


def decode_protocols_rich_text(value: object) -> str:
    if value in (None, "", [], {}):
        return ""

    if isinstance(value, dict):
        if is_draft_rich_text_doc(value):
            return render_draft_rich_text(value)
        return normalize_rich_text(json.dumps(value, ensure_ascii=False))

    if isinstance(value, list):
        rendered = [decode_protocols_rich_text(item) for item in value]
        return BLANKS_RE.sub("\n\n", "\n\n".join(item for item in rendered if item).strip())

    text = str(value)
    parsed = maybe_parse_json_text(text)
    if is_draft_rich_text_doc(parsed):
        return render_draft_rich_text(parsed)

    clean = html.unescape(text)
    clean = BREAK_RE.sub("\n", clean)
    clean = BLOCK_BREAK_RE.sub("\n", clean)
    clean = CELL_BREAK_RE.sub("\t", clean)
    clean = TAG_RE.sub(" ", clean)
    lines = [SPACE_RE.sub(" ", line).strip() for line in clean.splitlines()]
    return BLANKS_RE.sub("\n\n", "\n".join(line for line in lines if line)).strip()


def normalize_rich_text(text: str | None) -> str:
    return decode_protocols_rich_text(text)


def normalize_keywords(value: object) -> list[str]:
    raw_items: list[str] = []
    if value is None:
        return []
    if isinstance(value, str):
        raw_items.extend(KEYWORD_SPLIT_RE.split(value))
    elif isinstance(value, (list, tuple, set)):
        for item in value:
            raw_items.extend(normalize_keywords(item))
    elif isinstance(value, dict):
        for field in ("keyword", "name", "title", "value"):
            candidate = collapse_ws(str(value.get(field) or ""))
            if candidate:
                raw_items.append(candidate)
                break
    else:
        raw_items.append(str(value))

    normalized = [collapse_ws(item) for item in raw_items if collapse_ws(item)]
    return ordered_unique(normalized)


def strip_doi_prefix(value: str | None) -> str | None:

def fetch_api_payload(
    endpoint_template: str,
    identifier: str,
    headers: dict[str, str],
    timeout: int,
) -> tuple[dict[str, object], object | None, str]:
    url = endpoint_template.format(identifier=quote(identifier, safe=""))
    response = fetch(url, headers=headers, timeout=timeout)
    payload = None
    error = ""
    body = response.get("body", b"")
    if response.get("status") == 200:
        try:
            decoded = json.loads(body.decode("utf-8", "ignore"))
            payload = decoded.get("payload")
        except Exception as exc:
            error = f"json_decode_error:{exc!r}"
    elif response.get("status") == "ERROR":
        error = body.decode("utf-8", "ignore") or "network_error"
    else:
        error = f"http_{response.get('status')}"
    return response, payload, url if not error else url


def fetch_api_payload_with_retry(
    endpoint_template: str,
    identifier: str,
    headers: dict[str, str],
    timeout: int,
    limiter: RateLimiter,
    retry_limit: int,
    backoff_seconds: float,
) -> tuple[dict[str, object], object | None, str]:
    last_response: dict[str, object] | None = None
    last_payload: object | None = None
    last_url = endpoint_template.format(identifier=quote(identifier, safe=""))
    for attempt in range(1, max(retry_limit, 1) + 1):
        limiter.wait()
        response, payload, url = fetch_api_payload(endpoint_template, identifier, headers, timeout)
        last_response = response
        last_payload = payload
        last_url = url
        status = response.get("status")
        if status == 200 and payload is not None:
            return response, payload, url
        if status not in {429, 500, 502, 503, 504, "ERROR"} or attempt >= retry_limit:
            return response, payload, url
        time.sleep(backoff_seconds * attempt)
    assert last_response is not None
    return last_response, last_payload, last_url

def format_section_rows(steps_payload: list[dict[str, object]]) -> tuple[list[dict[str, object]], list[dict[str, object]]]:
    section_rows: list[dict[str, object]] = []
    step_rows: list[dict[str, object]] = []
    current_section_title = ""
    current_section_id: str | None = None
    section_index = 0
    for order_index, step in enumerate(steps_payload, start=1):
        section_title = decode_protocols_rich_text(step.get("section"))
        if section_title and section_title != current_section_title:
            section_index += 1
            current_section_id = f"section-{section_index:03d}"
            current_section_title = section_title
            section_rows.append(
                {
                    "section_id": current_section_id,
                    "order_index": section_index,
                    "title": section_title,
                }
            )
        step_text = decode_protocols_rich_text(step.get("step"))
        step_number = step.get("number") or order_index
        step_label = f"Step {step_number}"
        if step_text.startswith(step_label):
            normalized_text = step_text
        elif STEP_LABEL_RE.match(step_text):
            normalized_text = step_text
        else:
            normalized_text = f"{step_label}\n{step_text}".strip()
        step_rows.append(
            {
                "step_id": str(step.get("guid") or step.get("id") or order_index),
                "order_index": order_index,
                "step_number": str(step_number),
                "section_id": current_section_id,
                "section_title": current_section_title or None,
                "duration_seconds": int(step["duration"]) if isinstance(step.get("duration"), int) else None,
                "is_substep": bool(step.get("is_substep")),
                "critical_flags": [
                    collapse_ws(str(item.get("title") or item.get("icon") or ""))
                    for item in (step.get("critical") or [])
                    if isinstance(item, dict) and collapse_ws(str(item.get("title") or item.get("icon") or ""))
                ],
                "raw_text": str(step.get("step") or ""),
                "normalized_text": normalized_text,
            }
        )
    return section_rows, step_rows


def build_fulltext(protocol_payload: dict[str, object], step_rows: list[dict[str, object]]) -> str:
    parts: list[str] = []

    def append_block(title: str, value: str | None) -> None:
        clean = normalize_rich_text(value)
        if clean:
            parts.append(title)
            parts.append(clean)

    title = collapse_ws(str(protocol_payload.get("title") or ""))
    if title:
        parts.append(title)
    append_block("Description", decode_protocols_rich_text(protocol_payload.get("description")))
    append_block("Before You Begin", decode_protocols_rich_text(protocol_payload.get("before_start")))
    append_block("Guidelines", decode_protocols_rich_text(protocol_payload.get("guidelines")))
    append_block("Warnings", decode_protocols_rich_text(protocol_payload.get("warning")))
    append_block("Materials", decode_protocols_rich_text(protocol_payload.get("materials_text")))

    last_section = None
    for step in step_rows:
        section_title = step.get("section_title")
        if section_title and section_title != last_section:
            parts.append(f"Section: {section_title}")
            last_section = section_title
        parts.append(str(step["normalized_text"]))

    return BLANKS_RE.sub("\n\n", "\n\n".join(part for part in parts if part).strip())


def build_audit(
    protocol_payload: dict[str, object],
    section_rows: list[dict[str, object]],
    step_rows: list[dict[str, object]],
    fulltext: str,
) -> tuple[str, dict[str, object]]:
    warnings: list[str] = []
    nonempty_steps = sum(1 for row in step_rows if collapse_ws(str(row["normalized_text"])))
    if not collapse_ws(str(protocol_payload.get("title") or "")):
        warnings.append("missing_title")
    if not step_rows:
        warnings.append("no_steps")
    if nonempty_steps == 0:
        warnings.append("empty_steps")
    if not collapse_ws(str(protocol_payload.get("materials_text") or "")):
        warnings.append("missing_materials_text")
    if len(fulltext) < 200:
        warnings.append("short_fulltext")

    if not warnings:
        status = "ok"
    elif nonempty_steps > 0:
        status = "partial"
    else:
        status = "sparse"

    audit = {
        "warning_count": len(warnings),
        "warnings": warnings,
        "nonempty_step_count": nonempty_steps,
        "step_count": len(step_rows),
        "section_count": len(section_rows),
        "materials_char_count": len(decode_protocols_rich_text(protocol_payload.get("materials_text"))),
        "guidelines_char_count": len(decode_protocols_rich_text(protocol_payload.get("guidelines"))),
        "fulltext_char_count": len(fulltext),
    }
    return status, audit


def ensure_bundle_link(bundle_dir: Path, archive_root: Path, link_name: str, storage_relpath: str | None) -> None:
    link_path = bundle_dir / link_name
    if link_path.is_symlink() or link_path.is_file():
        link_path.unlink()
    elif link_path.is_dir():
        shutil.rmtree(link_path)
    if not storage_relpath:
        return
    target_path = archive_root / storage_relpath
    rel_target = os.path.relpath(target_path, bundle_dir)
    link_path.symlink_to(rel_target)

        "materials_text": decode_protocols_rich_text(protocol_payload.get("materials_text")) or None,

        "created_at": iso_from_timestamp(protocol_payload.get("created_on")),
        "modified_at": iso_from_timestamp(protocol_payload.get("modified_on")),
    }


def build_steps_bundle_payload(step_rows: list[dict[str, object]]) -> list[dict[str, object]]:
    return [
        {
            "step_id": row["step_id"],
            "order_index": row["order_index"],
            "step_number": row["step_number"],
            "section_id": row["section_id"],
            "section_title": row["section_title"],
            "duration_seconds": row["duration_seconds"],
            "is_substep": row["is_substep"],
            "critical_flags": row["critical_flags"],
            "text": row["normalized_text"],
        }
        for row in step_rows
    ]


def extract_protocol_metadata(protocol_payload: dict[str, object]) -> dict[str, object]:
    authors = [format_author_name(author) for author in (protocol_payload.get("authors") or []) if isinstance(author, dict)]
    authors = [author for author in authors if author]
    doi = strip_doi_prefix(str(protocol_payload.get("doi") or "")) or None
    version_uri = collapse_ws(str(protocol_payload.get("version_uri") or protocol_payload.get("uri") or ""))
    pdf_url = f"https://www.protocols.io/view/{version_uri}.pdf" if version_uri and protocol_payload.get("access", {}).get("can_download") else None
    return {
        "authors": authors,
        "keywords": normalize_keywords(protocol_payload.get("keywords")),
        "doi": doi,
        "pdf_url": pdf_url,
        "published_at": iso_from_timestamp(protocol_payload.get("published_on")),
        "version_uri": version_uri or None,
    }


def sync_record_bundle(
    archive_root: Path,
    protocol_id: str,
    version: int,
    manifest: dict[str, object],
    protocol_payload: dict[str, object],
    step_rows: list[dict[str, object]],
    authors: list[str],
    keywords: list[str],
    doi: str | None,
    pdf_url: str | None,
    protocol_blob: dict[str, object] | None,
    steps_blob: dict[str, object] | None,
    fulltext: str,
) -> str:
    relpath = bundle_relpath(protocol_id, version)
    bundle_dir = archive_root / relpath
    bundle_dir.mkdir(parents=True, exist_ok=True)
    legacy_manifest_dir = bundle_dir / "manifest.json"
    if legacy_manifest_dir.is_dir():
        shutil.rmtree(legacy_manifest_dir)

    write_json_file(bundle_dir / "manifest.json", manifest)
    write_json_file(
        bundle_dir / "protocol.json",
        build_protocol_bundle_payload(canonical_url=manifest["canonical_url"], protocol_payload=protocol_payload, authors=authors, keywords=keywords, doi=doi, pdf_url=pdf_url),
    )
    write_json_file(bundle_dir / "steps.json", build_steps_bundle_payload(step_rows))
    write_text_file(bundle_dir / "fulltext.txt", fulltext)
    ensure_bundle_link(bundle_dir, archive_root, "protocol.raw.json", protocol_blob.get("materialized_relpath") if protocol_blob else None)
    ensure_bundle_link(bundle_dir, archive_root, "steps.raw.json", steps_blob.get("materialized_relpath") if steps_blob else None)
    for legacy_name in ("protocol.json.gz", "steps.json.gz", "fulltext.txt.gz"):
        remove_bundle_path(bundle_dir / legacy_name)
    return str(relpath)


def record_api_fetch_attempt(
    conn: sqlite3.Connection,
    run_id: int,
    canonical_url: str,
    endpoint_kind: str,
    requested_url: str,
    response: dict[str, object],
    content_sha256: str | None,
    started_at: float,
    error: str | None,
) -> None:
    conn.execute(
        """
        INSERT INTO api_fetch_attempt(
            run_id, canonical_url, endpoint_kind, requested_url, http_status, content_type,
            byte_length, content_sha256, etag, last_modified, duration_ms, error
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """,
        (
            run_id,
            canonical_url,
            endpoint_kind,
            requested_url,
            response.get("status") if isinstance(response.get("status"), int) else None,
            response.get("content_type"),
            len(response.get("body", b"")),
            content_sha256,
            response.get("etag"),
            response.get("last_modified"),
            int((time.monotonic() - started_at) * 1000),
            error,
        ),
    )


def replace_child_rows_from_api(
    conn: sqlite3.Connection,
    canonical_url: str,
    protocol_payload: dict[str, object],
    section_rows: list[dict[str, object]],
    step_rows: list[dict[str, object]],
) -> dict[str, object]:
    lower_fulltext = "\n".join(
        [
            decode_protocols_rich_text(protocol_payload.get("materials_text")),
            decode_protocols_rich_text(protocol_payload.get("guidelines")),
            decode_protocols_rich_text(protocol_payload.get("warning")),
            *(row["normalized_text"] for row in step_rows),
        ]
    ).lower()
    meta = extract_protocol_metadata(protocol_payload)
    authors = list(meta["authors"])
    keywords = list(meta["keywords"])
    doi = meta["doi"]
    pdf_url = meta["pdf_url"]
    published_at = meta["published_at"]

    conn.execute("DELETE FROM protocol_author WHERE canonical_url = ?", (canonical_url,))
    conn.execute("DELETE FROM protocol_keyword WHERE canonical_url = ?", (canonical_url,))
    conn.execute("DELETE FROM protocol_section WHERE canonical_url = ?", (canonical_url,))
    conn.execute("DELETE FROM protocol_step WHERE canonical_url = ?", (canonical_url,))
    conn.execute("DELETE FROM protocol_asset WHERE canonical_url = ?", (canonical_url,))
    try:
        conn.execute("DELETE FROM protocol_search WHERE canonical_url = ?", (canonical_url,))
    except sqlite3.OperationalError:
        pass

    published_year = int(published_at[:4]) if published_at else None
    quality = {
        "source": "protocols_io_api",
        "has_title": bool(collapse_ws(str(protocol_payload.get("title") or ""))),
        "has_materials_text": bool(collapse_ws(str(protocol_payload.get("materials_text") or ""))),
        "has_guidelines": bool(collapse_ws(str(protocol_payload.get("guidelines") or ""))),
        "step_count": len(step_rows),
        "section_count": len(section_rows),
    }

    conn.execute(
        """
        INSERT INTO protocol_index(
            canonical_url, record_id, doi, protocols_io_integer_id, abstract, published_year,
            step_count, section_count, keyword_count, has_pdf, has_materials, has_reagents,
            has_equipment, has_troubleshooting, has_notes, is_protocol_like, quality_json, updated_at
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        ON CONFLICT(canonical_url) DO UPDATE SET
            record_id = excluded.record_id,
            doi = excluded.doi,
            protocols_io_integer_id = excluded.protocols_io_integer_id,
            abstract = excluded.abstract,
            published_year = excluded.published_year,
            step_count = excluded.step_count,
            section_count = excluded.section_count,
            keyword_count = excluded.keyword_count,
            has_pdf = excluded.has_pdf,
            has_materials = excluded.has_materials,
            has_reagents = excluded.has_reagents,
            has_equipment = excluded.has_equipment,
            has_troubleshooting = excluded.has_troubleshooting,
            has_notes = excluded.has_notes,
            is_protocol_like = excluded.is_protocol_like,
            quality_json = excluded.quality_json,
            updated_at = excluded.updated_at
        """,
        (
            canonical_url,
            collapse_ws(str(protocol_payload.get("uri") or "")) or None,
            doi,
            str(protocol_payload.get("id") or "") or None,
            decode_protocols_rich_text(protocol_payload.get("description")) or None,
            published_year,
            len(step_rows),
            len(section_rows),
            len(keywords),
            int(bool(pdf_url)),
            int("materials" in lower_fulltext),
            int("reagent" in lower_fulltext),
            int("equipment" in lower_fulltext),
            int("troubleshooting" in lower_fulltext),
            int(any(token in lower_fulltext for token in ["warning", "caution", "important", "note"])),
            int(len(step_rows) > 0),
            json_dumps(quality),
            now_utc(),
        ),
    )

    conn.executemany(
        "INSERT INTO protocol_author(canonical_url, author_order, author_name) VALUES (?, ?, ?)",
        [(canonical_url, order_index, author) for order_index, author in enumerate(authors, start=1)],
    )
    conn.executemany(
        "INSERT INTO protocol_keyword(canonical_url, keyword_order, keyword) VALUES (?, ?, ?)",
        [(canonical_url, order_index, keyword) for order_index, keyword in enumerate(keywords, start=1)],
    )
    conn.executemany(
        "INSERT INTO protocol_section(canonical_url, section_id, order_index, title) VALUES (?, ?, ?, ?)",
        [(canonical_url, row["section_id"], row["order_index"], row["title"]) for row in section_rows],
    )
    conn.executemany(
        """
        INSERT INTO protocol_step(
            canonical_url, step_id, order_index, section_id, section_title, raw_text, normalized_text
        ) VALUES (?, ?, ?, ?, ?, ?, ?)
        """,
        [
            (
                canonical_url,
                row["step_id"],
                row["order_index"],
                row["section_id"],
                row["section_title"],
                row["raw_text"],
                row["normalized_text"],
            )
            for row in step_rows
        ],
    )

    assets: list[tuple[str, str, str | None]] = []
    if doi:
        assets.append(("doi", doi, "protocol_api"))
    if pdf_url:
        assets.append(("pdf", pdf_url, "derived_from_version_uri"))
    if meta["version_uri"]:
        assets.append(("version_uri", meta["version_uri"], "protocol_api"))
    conn.executemany(
        "INSERT INTO protocol_asset(canonical_url, asset_type, url, note) VALUES (?, ?, ?, ?)",
        [(canonical_url, asset_type, url, note) for asset_type, url, note in assets],
    )

    try:
        conn.execute(
            """
            INSERT INTO protocol_search(canonical_url, title, description, authors, keywords, section_titles, step_text)
            VALUES (?, ?, ?, ?, ?, ?, ?)
            """,
            (
                canonical_url,
                collapse_ws(str(protocol_payload.get("title") or "")),
                decode_protocols_rich_text(protocol_payload.get("description")),
                "\n".join(authors),
                "\n".join(keywords),
                "\n".join(row["title"] for row in section_rows),
                "\n".join(row["normalized_text"] for row in step_rows),
            ),
        )
    except sqlite3.OperationalError:
        pass

    conn.execute(
        """
        UPDATE protocol_record
        SET title = ?,
            description = ?,
            published_at = ?,
            pdf_url = COALESCE(?, pdf_url),
            authors_json = ?,
            last_success_at = COALESCE(last_success_at, ?)
        WHERE canonical_url = ?
        """,
        (
            collapse_ws(str(protocol_payload.get("title") or "")) or None,
            decode_protocols_rich_text(protocol_payload.get("description")) or None,
            published_at,
            pdf_url,
            json_dumps(authors),
            now_utc(),
            canonical_url,
        ),
    )
    return {"authors": authors, "keywords": keywords, "doi": doi, "pdf_url": pdf_url}


def queue_records(conn: sqlite3.Connection, limit: int | None, force_refresh: bool) -> list[sqlite3.Row]:
    sql = """
    SELECT r.canonical_url, r.protocol_id, r.version, r.title
    FROM protocol_record AS r
    LEFT JOIN protocol_api_content AS a
      ON a.canonical_url = r.canonical_url
    WHERE r.protocol_id IS NOT NULL
    """
    params: list[object] = []
    if not force_refresh:
        sql += " AND (a.canonical_url IS NULL OR a.fetch_status NOT IN ('api_ready'))"
    sql += " ORDER BY COALESCE(r.sitemap_lastmod, '') DESC, r.protocol_id ASC, r.version ASC"
    if limit is not None:
        sql += " LIMIT ?"
        params.append(limit)
    return list(conn.execute(sql, params))


def fetch_record_payloads(
    row_data: dict[str, object],
    headers: dict[str, str],
    timeout: int,
    limiter: RateLimiter,
    retry_limit: int,
    backoff_seconds: float,
    fetch_delay_seconds: float,
) -> dict[str, object]:
    canonical_url = str(row_data["canonical_url"])
    identifier = api_identifier(str(row_data["protocol_id"]), int(row_data["version"]))

    protocol_started = time.monotonic()
    protocol_response, protocol_payload, protocol_url = fetch_api_payload_with_retry(
        API_PROTOCOL_URL_TEMPLATE,
        identifier,
        headers,
        timeout,
        limiter,
        retry_limit,
        backoff_seconds,
    )

    if fetch_delay_seconds > 0:
        time.sleep(fetch_delay_seconds)

    steps_started = time.monotonic()
    steps_response, steps_payload, steps_url = fetch_api_payload_with_retry(
        API_STEPS_URL_TEMPLATE,
        identifier,
        headers,
        timeout,
        limiter,
        retry_limit,
        backoff_seconds,
    )

    return {
        "row": row_data,
        "canonical_url": canonical_url,
        "protocol_started": protocol_started,
        "protocol_response": protocol_response,
        "protocol_payload": protocol_payload,
        "protocol_url": protocol_url,
        "steps_started": steps_started,
        "steps_response": steps_response,
        "steps_payload": steps_payload,
        "steps_url": steps_url,
    }


def process_record(
    conn: sqlite3.Connection,
    archive_root: Path,
    run_id: int,
    fetched: dict[str, object],
    materialize_uncompressed: bool,
) -> dict[str, object]:
    row = fetched["row"]
    canonical_url = str(fetched["canonical_url"])
    protocol_started = float(fetched["protocol_started"])
    protocol_response = fetched["protocol_response"]
    protocol_payload = fetched["protocol_payload"]
    protocol_url = str(fetched["protocol_url"])
    protocol_blob = None
    protocol_error = ""
    if protocol_payload is not None:
        protocol_blob = store_json_blob(
            archive_root,
            "protocol",
            protocol_payload,
            materialize_uncompressed=materialize_uncompressed,
        )
        ensure_json_blob_row(conn, protocol_blob)
    else:
        protocol_error = protocol_response.get("body", b"").decode("utf-8", "ignore")[:500]
    record_api_fetch_attempt(
        conn,
        run_id,
        canonical_url,
        "protocol",
        protocol_url,
        protocol_response,
        protocol_blob["sha256"] if protocol_blob else None,
        protocol_started,
        protocol_error or None,
    )
    steps_started = float(fetched["steps_started"])
    steps_response = fetched["steps_response"]
    steps_payload = fetched["steps_payload"]
    steps_url = str(fetched["steps_url"])
    steps_blob = None
    steps_error = ""
    if isinstance(steps_payload, list):
        steps_blob = store_json_blob(
            archive_root,
            "steps",
            steps_payload,
            materialize_uncompressed=materialize_uncompressed,
        )
        ensure_json_blob_row(conn, steps_blob)
    else:
        steps_error = steps_response.get("body", b"").decode("utf-8", "ignore")[:500]
    record_api_fetch_attempt(
        conn,
        run_id,
        canonical_url,
        "steps",
        steps_url,
        steps_response,
        steps_blob["sha256"] if steps_blob else None,
        steps_started,
        steps_error or None,
    )

    if not isinstance(protocol_payload, dict) or not isinstance(steps_payload, list):
        fetch_status = "api_error"
        audit_status = "error"
        audit = {
            "protocol_status": protocol_response.get("status"),
            "steps_status": steps_response.get("status"),
            "warnings": ["protocol_payload_missing" if not isinstance(protocol_payload, dict) else "", "steps_payload_missing" if not isinstance(steps_payload, list) else ""],
        }
        conn.execute(
            """
            INSERT INTO protocol_api_content(
                canonical_url, fetch_status, audit_status, protocol_http_status, steps_http_status,
                protocol_etag, steps_etag, protocol_last_modified, steps_last_modified,
                protocol_json_sha256, steps_json_sha256, audit_json, updated_at
            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            ON CONFLICT(canonical_url) DO UPDATE SET
                fetch_status = excluded.fetch_status,
                audit_status = excluded.audit_status,
                protocol_http_status = excluded.protocol_http_status,
                steps_http_status = excluded.steps_http_status,
                protocol_etag = excluded.protocol_etag,
                steps_etag = excluded.steps_etag,
                protocol_last_modified = excluded.protocol_last_modified,
                steps_last_modified = excluded.steps_last_modified,
                protocol_json_sha256 = excluded.protocol_json_sha256,
                steps_json_sha256 = excluded.steps_json_sha256,
                audit_json = excluded.audit_json,
                updated_at = excluded.updated_at
            """,
            (
                canonical_url,
                fetch_status,
                audit_status,
                protocol_response.get("status") if isinstance(protocol_response.get("status"), int) else None,
                steps_response.get("status") if isinstance(steps_response.get("status"), int) else None,
                protocol_response.get("etag"),
                steps_response.get("etag"),
                protocol_response.get("last_modified"),
                steps_response.get("last_modified"),
                protocol_blob["sha256"] if protocol_blob else None,
                steps_blob["sha256"] if steps_blob else None,
                json_dumps(audit),
                now_utc(),
            ),
        )
        return {"canonical_url": canonical_url, "status": fetch_status, "audit_status": audit_status}

    section_rows, step_rows = format_section_rows(steps_payload)
    fulltext = build_fulltext(protocol_payload, step_rows)
    text_blob = store_text_blob(
        archive_root,
        fulltext,
        "api_protocol_text",
        materialize_uncompressed=materialize_uncompressed,
    )
    ensure_text_blob_row(conn, text_blob)
    audit_status, audit = build_audit(protocol_payload, section_rows, step_rows, fulltext)
    extra = replace_child_rows_from_api(conn, canonical_url, protocol_payload, section_rows, step_rows)

    manifest = {
        "bundle_format_version": 2,
        "platform": "protocols_io",
        "canonical_url": canonical_url,
        "protocol_id": row["protocol_id"],
        "version": row["version"],
        "fetch_status": "api_ready",
        "audit_status": audit_status,
        "step_count": len(step_rows),
        "section_count": len(section_rows),
        "body_char_count": len(fulltext),
        "doi": extra["doi"],
        "pdf_url": extra["pdf_url"],
        "authors": extra["authors"],
        "keywords": extra["keywords"],
        "blobs": {
            "protocol_json": protocol_blob["storage_relpath"] if protocol_blob else None,
            "steps_json": steps_blob["storage_relpath"] if steps_blob else None,
            "fulltext_txt": text_blob["storage_relpath"],
        },
        "bundle_files": {
            "protocol_json": "protocol.json",
            "steps_json": "steps.json",
            "fulltext_txt": "fulltext.txt",
            "protocol_raw_json": "protocol.raw.json",
            "steps_raw_json": "steps.raw.json",
        },
        "audit": audit,
        "updated_at": now_utc(),
    }
    relpath = sync_record_bundle(
        archive_root,
        str(row["protocol_id"]),
        int(row["version"]),
        manifest,
        protocol_payload,
        step_rows,
        extra["authors"],
        extra["keywords"],
        extra["doi"],
        extra["pdf_url"],
        protocol_blob,
        steps_blob,
        fulltext,
    )

    conn.execute(
        """
        INSERT INTO protocol_api_content(
            canonical_url, fetch_status, audit_status, protocol_http_status, steps_http_status,
            protocol_etag, steps_etag, protocol_last_modified, steps_last_modified,
            protocol_json_sha256, steps_json_sha256, normalized_text_sha256, bundle_relpath,
            step_count, section_count, material_line_count, body_char_count, audit_json, updated_at
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        ON CONFLICT(canonical_url) DO UPDATE SET
            fetch_status = excluded.fetch_status,
            audit_status = excluded.audit_status,
            protocol_http_status = excluded.protocol_http_status,
            steps_http_status = excluded.steps_http_status,
            protocol_etag = excluded.protocol_etag,
            steps_etag = excluded.steps_etag,
            protocol_last_modified = excluded.protocol_last_modified,
            steps_last_modified = excluded.steps_last_modified,
            protocol_json_sha256 = excluded.protocol_json_sha256,
            steps_json_sha256 = excluded.steps_json_sha256,
            normalized_text_sha256 = excluded.normalized_text_sha256,
            bundle_relpath = excluded.bundle_relpath,
            step_count = excluded.step_count,
            section_count = excluded.section_count,
            material_line_count = excluded.material_line_count,
            body_char_count = excluded.body_char_count,
            audit_json = excluded.audit_json,
            updated_at = excluded.updated_at
        """,
        (
            canonical_url,
            "api_ready",
            audit_status,
            int(protocol_response["status"]),
            int(steps_response["status"]),
            protocol_response.get("etag"),
            steps_response.get("etag"),
            protocol_response.get("last_modified"),
            steps_response.get("last_modified"),
            protocol_blob["sha256"] if protocol_blob else None,
            steps_blob["sha256"] if steps_blob else None,
            text_blob["sha256"],
            relpath,
            len(step_rows),
            len(section_rows),
            len(decode_protocols_rich_text(protocol_payload.get("materials_text")).splitlines()),
            len(fulltext),
            json_dumps(audit),
            now_utc(),
        ),
    )
    return {
        "canonical_url": canonical_url,
        "status": "api_ready",
        "audit_status": audit_status,
        "step_count": len(step_rows),
        "body_char_count": len(fulltext),
        "bundle_relpath": relpath,
    }


def summary(conn: sqlite3.Connection) -> dict[str, int]:
    return dict(
        conn.execute(
            """
            SELECT
                (SELECT COUNT(*) FROM protocol_api_content) AS api_rows,
                (SELECT COUNT(*) FROM protocol_api_content WHERE fetch_status = 'api_ready') AS api_ready_rows,
                (SELECT COUNT(*) FROM protocol_api_content WHERE audit_status = 'ok') AS api_audit_ok_rows,
                (SELECT COUNT(*) FROM protocol_api_content WHERE audit_status = 'partial') AS api_audit_partial_rows,
                (SELECT COUNT(*) FROM json_blob) AS json_blobs,
                (SELECT COUNT(*) FROM api_fetch_attempt) AS api_fetch_attempts
            """
        ).fetchone()
    )


def run_batch(
    conn: sqlite3.Connection,
    archive_root: Path,
    run_id: int,
    headers: dict[str, str],
    args: argparse.Namespace,
) -> tuple[int, int, list[dict[str, object]], dict[str, int]]:
    rows = queue_records(conn, args.max_records, args.force_refresh)
    if not rows:
        return 0, 0, [], summary(conn)

    limiter = RateLimiter(args.max_rps)
    sampler = ReservoirSampler(size=args.audit_sample_size, seed=7)
    processed = 0
    workers = max(1, args.workers)
    row_dicts = [dict(row) for row in rows]

    with ThreadPoolExecutor(max_workers=workers, thread_name_prefix="protocols-io-api") as executor:
        future_to_row: dict[Future[dict[str, object]], dict[str, object]] = {}

        def submit(row_data: dict[str, object]) -> Future[dict[str, object]]:
            return executor.submit(
                fetch_record_payloads,
                row_data,
                headers,
                args.timeout,
                limiter,
                args.retry_limit,
                args.backoff_seconds,
                args.fetch_delay_seconds,
            )

        preload = min(workers, len(row_dicts))
        next_index = 0
        for _ in range(preload):
            row_data = row_dicts[next_index]
            future_to_row[submit(row_data)] = row_data
            next_index += 1

        while future_to_row:
            done, _ = wait(list(future_to_row.keys()), return_when=FIRST_COMPLETED)
            for future in done:
                row_data = future_to_row.pop(future)
                fetched = future.result()
                result = process_record(
                    conn,
                    archive_root,
                    run_id,
                    fetched,
                    args.materialize_uncompressed,
                )
                processed += 1
                sampler.consider(result)
                conn.commit()
                print(
                    f"[protocols.io] api {processed}/{len(row_dicts)} status={result['status']} "
                    f"audit={result.get('audit_status')} url={row_data['canonical_url']}"
                )
                if next_index < len(row_dicts):
                    next_row = row_dicts[next_index]
                    future_to_row[submit(next_row)] = next_row
                    next_index += 1

    return len(row_dicts), processed, sampler.items, summary(conn)


def main() -> int:
    if hasattr(sys.stdout, "reconfigure"):
        sys.stdout.reconfigure(encoding="utf-8", errors="replace")

    args = parse_args()
    access_token = os.environ.get(args.access_token_env, "").strip()
    if not access_token:
        raise SystemExit(f"missing access token in env var {args.access_token_env}")

    archive_root = Path(args.archive_root).resolve()
    ensure_archive_layout(archive_root)
    export_schema_copy(archive_root)
    db_path = archive_root / "db" / "protocols_io_raw.sqlite3"

    ensure_schema(conn)

    headers = {"Authorization": f"Bearer {access_token}", "Accept": "application/json"}
    try:
        while True:
            run_id = start_run(conn, mode="archive_api", note=args.note or None)
            try:
                queued, processed, audit_sample, stats = run_batch(conn, archive_root, run_id, headers, args)
                payload = {
                    "archive_root": str(archive_root),
                    "db_path": str(db_path),
                    "processed": processed,
                    "queued": queued,

                    "audit_sample": audit_sample,
                    "user_agent": USER_AGENT,
                    "note": args.note,
                    "workers": args.workers,
                    "max_rps": args.max_rps,
                }
                finish_run(conn, run_id, "completed", payload)
                conn.commit()
                write_manifest(archive_root, payload)
                print(json.dumps(payload, ensure_ascii=False, indent=2))
            except Exception as exc:
                payload = {
                    "archive_root": str(archive_root),
                    "db_path": str(db_path),
                    "summary": summary(conn),
                    "error": repr(exc),
                    "note": args.note,
                    "workers": args.workers,
                    "max_rps": args.max_rps,
                }
                finish_run(conn, run_id, "failed", payload)
                conn.commit()
                write_manifest(archive_root, payload)
                raise

            if not args.loop_forever:
                break
            if payload["processed"] == 0:
                print(f"[protocols.io] api idle; sleeping {args.idle_sleep_seconds:.0f}s before next resume check")
                time.sleep(args.idle_sleep_seconds)
    finally:
        conn.close()
    return 0
