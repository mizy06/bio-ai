数据来源与覆盖领域

来自 Bio-protocol 官方站点。公开 sitemap 已明确拆出 protocol、Bio101、RAPProtocol、collection 和 conciseprotocol 等对象层，覆盖实验方法正文、教学型方法说明、快速协议与专题集合。

数据规模与更新情况

站点主页、robots.txt 与 sitemap.xml 均可复核。当前 sitemap 索引含 855 个子 sitemap，其中主 protocol 层 4,867 个 detail URL，Bio101 1,025 个，RAPProtocol 2,410 个，collection 46 个，另有 847 个 conciseprotocol 分片 sitemap。

下载单元数量

若按湿实验主正文层统计，主 protocol 层当前可核 4,867 个 detail URL；若连同 Bio101、RAP 与 collection 一并计入，则对象层明显大于单一 article 口径。

单个下载单元平均大小

正文 detail 页在当前环境下返回 Safeline 访问校验页，未将挑战页 HTML 大小误记为正文单元均值。

数据格式与字段结构

主 detail URL 形态为 bpdetail?id=...&type=0。公开结构可落成 protocol article → section / figures / tables / references，并带有 Bio101、RAPProtocol、collection 等并列内容层。

数据内容特点与适用场景

一方面可提供较标准的 article 式 protocol 正文，另一方面站内还包含配方、准备、教学说明和快速协议，适合补 protocols.io 之外的 article 风格协议语料。

获取方式、开放性与可用性

主页、归档页、robots 与 sitemap 均可公开访问，适合批量发现；英文 detail 页在当前环境下返回访问校验页，因此发现层强、正文层需另行处理。

def store_blob(archive_root: Path, namespace: str, body: bytes, extension: str) -> dict[str, Any]:
    sha256 = hashlib.sha256(body).hexdigest()
    relpath = Path("objects") / namespace / f"{sha256}{extension}.gz"
    abspath = archive_root / relpath
    compressed = gzip.compress(body, compresslevel=6)
    if not abspath.exists():
        abspath.parent.mkdir(parents=True, exist_ok=True)
        abspath.write_bytes(compressed)
    return {
        "sha256": sha256,
        "storage_relpath": str(relpath),
        "byte_length": len(body),
        "gzip_length": len(compressed),
    }

class RateLimiter:
    def __init__(self, max_rps: float) -> None:
        self.interval = 0.0 if max_rps <= 0 else 1.0 / max_rps
        self.lock = threading.Lock()
        self.next_allowed_at = 0.0

    def wait(self) -> None:
        if self.interval <= 0:
            return
        while True:
            with self.lock:
                now = time.monotonic()
                if now >= self.next_allowed_at:
                    self.next_allowed_at = now + self.interval
                    return
                sleep_for = self.next_allowed_at - now
            time.sleep(sleep_for)


def http_get(url: str, timeout: int) -> dict[str, Any]:
    return fetch(url, headers=build_http_headers(), timeout=timeout)


def fetch_with_retry(
    url: str,
    timeout: int,
    limiter: RateLimiter,
    retry_limit: int,
    backoff_seconds: float,
) -> dict[str, Any]:
    last_response: dict[str, Any] | None = None
    attempts = max(retry_limit, 1)
    for attempt in range(1, attempts + 1):
        if STOP_EVENT.is_set():
            return last_response or {"url": url, "status": "ERROR", "body": b"", "final_url": url, "content_type": ""}
        limiter.wait()
        response = http_get(url, timeout=timeout)
        last_response = response
        status = response.get("status")
        if status == 200:
            return response
        if status not in RETRYABLE_STATUSES or attempt >= attempts:
            return response
        time.sleep(backoff_seconds * attempt)
    return last_response or {"url": url, "status": "ERROR", "body": b""}


def is_pdf_response(response: dict[str, Any]) -> bool:
    if response.get("status") != 200:
        return False
    content_type = str(response.get("content_type") or "").lower()
    body = response.get("body", b"")
    return "application/pdf" in content_type or (isinstance(body, bytes) and body.startswith(b"%PDF-"))

def discover_protocols(
    conn: sqlite3.Connection,
    archive_root: Path,
    timeout: int,
    limiter: RateLimiter,
    retry_limit: int,
    backoff_seconds: float,
) -> tuple[int, int]:
    index_response = fetch_with_retry(SITEMAP_INDEX_URL, timeout, limiter, retry_limit, backoff_seconds)
    sitemap_urls: list[str] = []
    if index_response.get("status") == 200:
        sitemap_urls = [url for url in iter_sitemap_index(index_response.get("body", b"")) if url == PROTOCOL_SITEMAP_URL]
    record_discovery_snapshot(
        conn,
        archive_root,
        SITEMAP_INDEX_URL,
        "sitemap_index",
        index_response,
        len(sitemap_urls),
        0,
        0,
        None if index_response.get("status") == 200 else "http_error",
    )
    if not sitemap_urls:
        return 0, 0

    total_new_records = 0
    total_new_variants = 0
    for sitemap_url in sitemap_urls:
        response = fetch_with_retry(sitemap_url, timeout, limiter, retry_limit, backoff_seconds)
        entries = iter_sitemap_entries(response.get("body", b"")) if response.get("status") == 200 else []
        new_records, new_variants = upsert_discovery_batch(conn, entries, sitemap_url)
        total_new_records += new_records
        total_new_variants += new_variants
        record_discovery_snapshot(
            conn,
            archive_root,
            sitemap_url,
            "protocol_sitemap",
            response,
            len(entries),
            new_variants,
            new_records,
            None if response.get("status") == 200 else "http_error",
        )
    return total_new_records, total_new_variants


def upsert_discovery_batch(
    conn: sqlite3.Connection,
    entries: list[dict[str, Any]],
    sitemap_url: str,
) -> tuple[int, int]:
    variants: list[dict[str, Any]] = []
    for entry in entries:
        loc = str(entry.get("loc") or "").strip()
        article_id = article_id_from_url(loc)
        if article_id is None:
            continue
        variants.append(
            {
                "variant_url": loc,
                "canonical_url": canonical_detail_url(article_id),
                "pdf_url": pdf_url_for_article(article_id),
                "article_id": article_id,
                "record_key": record_key_for_article(article_id),
                "locale": locale_from_url(loc),
                "lastmod": entry.get("lastmod"),
                "source_url": sitemap_url,
            }
        )
    if not variants:
        return 0, 0

    canonical_urls = sorted({row["canonical_url"] for row in variants})
    existing_canonicals = {
        row["canonical_url"]
        for row in conn.execute(
            f"SELECT canonical_url FROM article_record WHERE canonical_url IN ({','.join('?' for _ in canonical_urls)})",
            canonical_urls,
        )
    }
    variant_urls = [row["variant_url"] for row in variants]
    existing_variants = {
        row["variant_url"]
        for row in conn.execute(
            f"SELECT variant_url FROM article_variant WHERE variant_url IN ({','.join('?' for _ in variant_urls)})",
            variant_urls,
        )
    }

    by_canonical: dict[str, dict[str, Any]] = {}
    for variant in variants:
        row = by_canonical.setdefault(
            variant["canonical_url"],
            {
                "canonical_url": variant["canonical_url"],
                "article_id": variant["article_id"],
                "record_key": variant["record_key"],
                "pdf_url": variant["pdf_url"],
                "latest_variant_url": variant["variant_url"],
                "locale_primary": variant["locale"],
                "sitemap_lastmod": variant["lastmod"],
                "variant_count": 0,
            },
        )
        row["variant_count"] += 1
        row["latest_variant_url"] = variant["variant_url"]
        if variant["locale"]:
            row["locale_primary"] = variant["locale"]
        if variant["lastmod"] and (not row["sitemap_lastmod"] or variant["lastmod"] > row["sitemap_lastmod"]):
            row["sitemap_lastmod"] = variant["lastmod"]

    discovered_at = now_utc()
    for row in by_canonical.values():
        meta_json = {
            "article_id": row["article_id"],
            "pdf_url": row["pdf_url"],
            "latest_variant_url": row["latest_variant_url"],
            "sitemap_lastmod": row["sitemap_lastmod"],
            "locale_primary": row["locale_primary"],
        }
        conn.execute(
            """
            INSERT INTO article_record(
                canonical_url, platform, article_id, record_key, latest_variant_url, pdf_url, locale_primary,
                status, discovery_source, variant_count, sitemap_lastmod, first_discovered_at, last_discovered_at, meta_json
            ) VALUES (?, ?, ?, ?, ?, ?, ?, 'discovered', 'protocol_sitemap', ?, ?, ?, ?, ?)
            ON CONFLICT(canonical_url) DO UPDATE SET
                latest_variant_url = excluded.latest_variant_url,
                pdf_url = excluded.pdf_url,
                locale_primary = COALESCE(excluded.locale_primary, article_record.locale_primary),
                discovery_source = excluded.discovery_source,
                variant_count = excluded.variant_count,
                sitemap_lastmod = COALESCE(excluded.sitemap_lastmod, article_record.sitemap_lastmod),
                last_discovered_at = excluded.last_discovered_at,
                meta_json = excluded.meta_json
            """,
            (
                row["canonical_url"],
                PLATFORM_NAME,
                row["article_id"],
                row["record_key"],
                row["latest_variant_url"],
                row["pdf_url"],
                row["locale_primary"],
                row["variant_count"],
                row["sitemap_lastmod"],
                discovered_at,
                discovered_at,
                json_dumps(meta_json),
            ),
        )

    for variant in variants:
        conn.execute(
            """
            INSERT INTO article_variant(variant_url, canonical_url, locale, source_url, lastmod)
            VALUES (?, ?, ?, ?, ?)
            ON CONFLICT(variant_url) DO UPDATE SET
                canonical_url = excluded.canonical_url,
                locale = excluded.locale,
                source_url = excluded.source_url,
                lastmod = excluded.lastmod
            """,
            (
                variant["variant_url"],
                variant["canonical_url"],
                variant["locale"],
                variant["source_url"],
                variant["lastmod"],
            ),
        )

    return len(canonical_urls) - len(existing_canonicals), len(variant_urls) - len(existing_variants)


def select_fetch_queue(conn: sqlite3.Connection, limit: int | None, force_refresh: bool) -> list[sqlite3.Row]:
    sql = """
    SELECT canonical_url, article_id, record_key, latest_variant_url, pdf_url
    FROM article_record
    WHERE (
            ? = 1
         OR status IN ('discovered', 'fetch_error')
      )
    ORDER BY COALESCE(sitemap_lastmod, '') DESC, article_id DESC
    """
    params: list[Any] = [1 if force_refresh else 0]
    if limit is not None:
        sql += " LIMIT ?"
        params.append(limit)
    return list(conn.execute(sql, params))


def fetch_pdf_task(
    row_data: dict[str, Any],
    timeout: int,
    limiter: RateLimiter,
    retry_limit: int,
    backoff_seconds: float,
) -> dict[str, Any]:
    requested_url = str(row_data["pdf_url"])
    started = time.monotonic()
    response = fetch_with_retry(requested_url, timeout, limiter, retry_limit, backoff_seconds)
    duration_ms = int((time.monotonic() - started) * 1000)
    return {
        "row": row_data,
        "requested_url": requested_url,
        "response": response,
        "duration_ms": duration_ms,
    }


def process_article(

    duration_ms: int,

    blob: dict[str, Any] | None = None
    error: str | None = None
    if is_pdf_response(response):
        blob = store_blob(archive_root, "pdf", response.get("body", b""), ".pdf")
        ensure_pdf_blob_row(conn, blob)
        meta_json = {
            "article_id": row["article_id"],
            "detail_url": row["canonical_url"],
            "variant_url": row["latest_variant_url"],
            "pdf_url": requested_url,
            "final_pdf_url": response.get("final_url"),
            "etag": response.get("etag"),
            "last_modified": response.get("last_modified"),
        }
        conn.execute(
            """
            UPDATE article_record
            SET status = 'archived',
                last_fetched_at = ?,
                last_success_at = ?,
                last_http_status = 200,
                last_error = NULL,
                content_sha256 = ?,
                meta_json = ?
            WHERE canonical_url = ?
            """,
            (now_utc(), now_utc(), blob["sha256"], json_dumps(meta_json), row["canonical_url"]),
        )
        write_record_bundle(
            archive_root,
            str(row["record_key"]),
            str(blob["storage_relpath"]),
            {
                "platform": PLATFORM_NAME,
                "record_key": row["record_key"],
                "article_id": row["article_id"],
                "canonical_url": row["canonical_url"],
                "variant_url": row["latest_variant_url"],
                "pdf_url": requested_url,
                "final_pdf_url": response.get("final_url"),
                "content_sha256": blob["sha256"],
                "storage_relpath": blob["storage_relpath"],
                "generated_at": now_utc(),
            },
        )
    else:
        status = response.get("status")
        record_status = "fetch_error"
        if status == 200:
            error = "non_pdf_response"
            record_status = "no_pdf_available"
        elif status == "ERROR":
            error = "network_error"
        else:
            error = f"http_{status}"
        conn.execute(
            """
            UPDATE article_record
            SET status = ?,
                last_fetched_at = ?,
                last_http_status = ?,
                last_error = ?
            WHERE canonical_url = ?
            """,
            (record_status, now_utc(), status if isinstance(status, int) else None, error, row["canonical_url"]),
        )

    record_fetch_attempt(conn, run_id, row["canonical_url"], requested_url, response, blob, duration_ms, error)
    return {
        "canonical_url": row["canonical_url"],
        "article_id": row["article_id"],
        "http_status": response.get("status"),
        "error": error,
        "byte_length": len(response.get("body", b"")),
    }


def run_fetch_batch(
    conn: sqlite3.Connection,
    archive_root: Path,
    run_id: int,
    rows: list[sqlite3.Row],
    timeout: int,
    workers: int,
    max_rps: float,
    retry_limit: int,
    backoff_seconds: float,
) -> tuple[int, int]:
    if not rows:
        return 0, 0
    limiter = RateLimiter(max_rps)
    row_dicts = [dict(row) for row in rows]
    archived = 0
    errors = 0
    processed = 0
    workers = max(1, workers)

    with ThreadPoolExecutor(max_workers=workers, thread_name_prefix="bio-protocol-pdf") as executor:
        future_to_row: dict[Future[dict[str, Any]], dict[str, Any]] = {}

        def submit(row_data: dict[str, Any]) -> Future[dict[str, Any]]:
            return executor.submit(fetch_pdf_task, row_data, timeout, limiter, retry_limit, backoff_seconds)

        next_index = 0
        preload = min(workers, len(row_dicts))
        for _ in range(preload):
            if STOP_EVENT.is_set():
                break
            row_data = row_dicts[next_index]
            future_to_row[submit(row_data)] = row_data
            next_index += 1

        while future_to_row:
            done, _ = wait(list(future_to_row.keys()), return_when=FIRST_COMPLETED)
            for future in done:
                row_data = future_to_row.pop(future)
                fetched = future.result()
                result = process_article(conn, archive_root, run_id, row_data, fetched["response"], int(fetched["duration_ms"]))
                processed += 1
                if result["http_status"] == 200 and not result["error"]:
                    archived += 1
                else:
                    errors += 1
                conn.commit()
                print(
                    f"[{PLATFORM_NAME}] pdf {processed}/{len(row_dicts)} "
                    f"status={result['http_status']} error={result.get('error') or 'ok'} "
                    f"article_id={row_data['article_id']} url={row_data['pdf_url']}"
                )
                if STOP_EVENT.is_set():
                    for pending in future_to_row:
                        pending.cancel()
                    future_to_row.clear()
                    break
                if next_index < len(row_dicts):
                    next_row = row_dicts[next_index]
                    future_to_row[submit(next_row)] = next_row
                    next_index += 1
    return archived, errors


def archive_summary(conn: sqlite3.Connection) -> dict[str, Any]:
    return dict(
        conn.execute(
            """
            SELECT
                (SELECT COUNT(*) FROM article_record) AS records,
                (SELECT COUNT(*) FROM article_record WHERE status = 'archived') AS archived_records,
                (SELECT COUNT(*) FROM article_record WHERE status = 'fetch_error') AS error_records,
                (SELECT COUNT(*) FROM article_record WHERE status = 'no_pdf_available') AS no_pdf_records,
                (SELECT COUNT(*) FROM article_variant) AS variants,
                (SELECT COUNT(*) FROM pdf_blob) AS pdf_blobs,
                (SELECT COUNT(*) FROM fetch_attempt) AS fetch_attempts
            """
        ).fetchone()
    )


def compute_effective_max_rps(robots_crawl_delay: float | None, requested_max_rps: float | None) -> float:
    robots_max_rps = (1.0 / robots_crawl_delay) if robots_crawl_delay and robots_crawl_delay > 0 else None
    if requested_max_rps is None:
        return robots_max_rps or FALLBACK_MAX_RPS
    if robots_max_rps is None:
        return requested_max_rps
    return min(requested_max_rps, robots_max_rps)


def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(description="Archive Bio-Protocol protocol PDFs into a raw SQLite + content-addressed store.")
    parser.add_argument("--archive-root", default=str(DEFAULT_ARCHIVE_ROOT))
    parser.add_argument("--timeout", type=int, default=DEFAULT_TIMEOUT)
    parser.add_argument("--workers", type=int, default=DEFAULT_WORKERS)
    parser.add_argument("--max-rps", type=float, default=None, help="Optional ceiling; values above robots crawl-delay are clamped.")
    parser.add_argument("--retry-limit", type=int, default=DEFAULT_RETRY_LIMIT)
    parser.add_argument("--backoff-seconds", type=float, default=DEFAULT_BACKOFF_SECONDS)
    parser.add_argument("--loop-forever", action="store_true")
    parser.add_argument("--idle-sleep-seconds", type=float, default=DEFAULT_IDLE_SLEEP_SECONDS)
    parser.add_argument("--max-records", type=int, default=None)
    parser.add_argument("--force-refresh", action="store_true")
    parser.add_argument("--skip-discovery", action="store_true")
    parser.add_argument("--note", default="")
    return parser.parse_args()


def run_archive(args: argparse.Namespace) -> dict[str, Any]:
    archive_root = Path(args.archive_root).expanduser().resolve()
    ensure_archive_layout(archive_root)
    db_path = archive_root / "db" / f"{PLATFORM_NAME}.sqlite3"
    conn = connect_db(db_path)
    ensure_schema(conn)
    export_schema_copy(archive_root)

    robots_response = http_get(ROBOTS_URL, timeout=args.timeout)
    robots_summary = None
    effective_max_rps = args.max_rps if args.max_rps is not None else FALLBACK_MAX_RPS
    if robots_response.get("status") == 200:
        robots_text = robots_response.get("body", b"").decode("utf-8", "ignore")
        robots_summary = record_robots_snapshot(conn, robots_text)
        effective_max_rps = compute_effective_max_rps(robots_summary["crawl_delay_seconds"], args.max_rps)
    conn.commit()

    run_id = start_run(conn, mode="pdf_archive", note=args.note or None)
    conn.commit()

    discovered_records = 0
    discovered_variants = 0
    processed_total = 0
    archived_total = 0
    errors_total = 0
    first_pass = True
    queue: list[sqlite3.Row] = []

    try:
        while True:
            new_records = 0
            new_variants = 0
            if not args.skip_discovery and (first_pass or not queue):
                limiter = RateLimiter(effective_max_rps)
                new_records, new_variants = discover_protocols(
                    conn,
                    archive_root,
                    args.timeout,
                    limiter,
                    args.retry_limit,
                    args.backoff_seconds,
                )
                conn.commit()
                discovered_records += new_records
                discovered_variants += new_variants

            queue = select_fetch_queue(conn, args.max_records, args.force_refresh)
            archived_this_pass, errors_this_pass = run_fetch_batch(
                conn,
                archive_root,
                run_id,
                queue,
                args.timeout,
                args.workers,
                effective_max_rps,
                args.retry_limit,
                args.backoff_seconds,
            )
            processed_total += len(queue)
            archived_total += archived_this_pass
            errors_total += errors_this_pass

            summary = archive_summary(conn)
            summary.update(
                run_id=run_id,
                archive_root=str(archive_root),
                db_path=str(db_path),
                workers=args.workers,
                max_rps_effective=effective_max_rps,
                discovered_records_this_run=discovered_records,
                discovered_variants_this_run=discovered_variants,
                processed_this_run=processed_total,
                archived_this_run=archived_total,
                errors_this_run=errors_total,
                robots=robots_summary,
            )
            write_manifest(archive_root, summary)

            if STOP_EVENT.is_set():
                finish_run(conn, run_id, "interrupted", summary)
                conn.commit()
                return summary

            if not args.loop_forever:
                finish_run(conn, run_id, "completed", summary)
                conn.commit()
                return summary

            if not queue and new_records == 0:
                time.sleep(args.idle_sleep_seconds)
            first_pass = False
    except KeyboardInterrupt:
        summary = archive_summary(conn)
        summary.update(
            run_id=run_id,
            archive_root=str(archive_root),
            db_path=str(db_path),
            workers=args.workers,
            max_rps_effective=effective_max_rps,
            discovered_records_this_run=discovered_records,
            discovered_variants_this_run=discovered_variants,
            processed_this_run=processed_total,
            archived_this_run=archived_total,
            errors_this_run=errors_total,
            robots=robots_summary,
        )
        finish_run(conn, run_id, "interrupted", summary)
        conn.commit()
        write_manifest(archive_root, summary)
        raise


def main() -> int:
    signal.signal(signal.SIGTERM, handle_termination_signal)
    signal.signal(signal.SIGINT, handle_termination_signal)
    args = parse_args()
    summary = run_archive(args)
    print(json.dumps(summary, ensure_ascii=False, indent=2))
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
