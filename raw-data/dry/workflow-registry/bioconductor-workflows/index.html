爬取脚本
from pathlib import Path
import re
import sys

sys.path.append(str(Path(__file__).resolve().parents[1]))
from common import fetch_gzip_text, fetch_text, save_snapshot


def main() -> None:
    landing = fetch_text("https://bioconductor.org/packages/release/workflows/")
    packages = fetch_gzip_text("https://bioconductor.org/packages/release/bioc/src/contrib/PACKAGES.gz")
    names = re.findall(r"^Package: (.+)$", packages.get("text", ""), flags=re.M)[:10]
    payload = {
        "landing_preview": landing.get("text", "")[:200],
        "sample_packages": names,
    }
    print(save_snapshot("dry/bioconductor_workflows.json", payload))


if __name__ == "__main__":
    main()
