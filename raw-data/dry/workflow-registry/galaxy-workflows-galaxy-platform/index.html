爬取脚本
from pathlib import Path
import sys

sys.path.append(str(Path(__file__).resolve().parents[1]))
from common import fetch_json, save_snapshot


def main() -> None:
    version = fetch_json("https://usegalaxy.org/api/version")
    workflows = fetch_json("https://usegalaxy.org/api/workflows?limit=1")
    payload = {
        "instance_version": (version.get("json") or {}).get("version_major"),
        "workflow_count": len(workflows.get("json") or []),
        "sample_workflow_ids": [item.get("id") for item in (workflows.get("json") or [])[:3]],
    }
    print(save_snapshot("dry/galaxy_workflows.json", payload))


if __name__ == "__main__":
    main()
