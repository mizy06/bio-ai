最小爬取伪代码

1. 先读取 `robots.txt` 明确公开页面与 `/api/` 的访问边界。
2. 把 `developers` 页作为 API 授权与字段语义说明源。
3. 公开抓取阶段只下钻 public protocol 页面，不伪造 Bearer token 调私有 API。
4. 从 protocol 页面抽取 `protocol -> version -> step`、materials、references 和 DOI。
5. 附件与下载资源单独记为 asset 层，不和正文 step 混写。
爬取脚本
from pathlib import Path
import re
import sys

sys.path.append(str(Path(__file__).resolve().parents[1]))
from common import fetch_text, save_snapshot


def main() -> None:
    robots = fetch_text("https://www.protocols.io/robots.txt")
    dev = fetch_text("https://www.protocols.io/developers")
    payload = {
        "api_disallow_lines": [line for line in robots.get("text", "").splitlines() if "Disallow:" in line and "/api/" in line],
        "developer_page_has_oauth": bool(re.search(r"OAuth|Bearer", dev.get("text", ""), re.I)),
        "developer_page_preview": dev.get("text", "")[:300],
    }
    print(save_snapshot("wet/protocols_io.json", payload))


if __name__ == "__main__":
    main()
